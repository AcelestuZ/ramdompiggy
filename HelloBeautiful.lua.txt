local function loadPart(url, name)
    local success, result = pcall(function()
        return game:HttpGet(url)
    end)
    if success and result then
        local func, err = loadstring(result)
        if func then
            func()
            print("[NLT] " .. name .. " loaded ")
        else
            warn("[NLT] Error in " .. name .. ": " .. tostring(err))
        end
    else
        warn("[NLT] Not valid Link " .. name)
    end
end
