local function loadPart(url, name)
    local success, err = pcall(function()
        loadstring(game:HttpGet(url))()
    end)
    if success then
        print("[NLT] " .. name .. " caricata con successo.")
    else
        warn("[NLT] Errore nel caricamento di " .. name .. ": " .. err)
    end
end
