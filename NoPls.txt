local lp = game:GetService("Players").LocalPlayer
local lt = game:GetService("Lighting")
local rs = game:GetService("RunService")
local active = false

local screenGui = Instance.new("ScreenGui")
screenGui.Name = "AntiLoveUI"
screenGui.Parent = lp.PlayerGui
screenGui.ResetOnSpawn = false

local btn = Instance.new("TextButton")
btn.Size = UDim2.new(0, 150, 0, 50)
btn.Position = UDim2.new(0, 50, 0, 110) -- Posizionato sotto gli altri per non sovrapporsi
btn.BackgroundColor3 = Color3.fromRGB(200, 0, 100)
btn.Text = "NoLoveTrap: OFF"
btn.TextColor3 = Color3.fromRGB(255, 255, 255)
btn.Font = Enum.Font.SourceSansBold
btn.TextSize = 16
btn.Active = true
btn.Draggable = true 
btn.Parent = screenGui

local function cleanLove()
	if not active then return end
	
	-- 1. Anti-Nebbia (spesso diventa rosa/fucsia)
	lt.FogEnd = 100000
	
	-- 2. Anti-GUI Effects (Cuori, Vignette, Gradienti)
	local guis = {lp.PlayerGui:FindFirstChild("MainGui"), lp.PlayerGui:FindFirstChild("GameGUI")}
	for _, gui in pairs(guis) do
		if gui then
			for _, v in pairs(gui:GetDescendants()) do
				-- Target specifici per Lovestruck e Inversione colori
				if v.Name == "LovestruckEffect" or v.Name == "HeartVignette" or 
				   v.Name == "Vignette" or v.Name:find("Invert") or v.Name:find("Pink") then
					
					if v:IsA("Frame") or v:IsA("ScrollingFrame") then
						v.BackgroundTransparency = 1
						v.Visible = true
					elseif v:IsA("ImageLabel") or v:IsA("ImageButton") then
						v.ImageTransparency = 1
						v.BackgroundTransparency = 1
						v.Visible = true
					elseif v:IsA("UIGradient") then
						v.Enabled = false -- Rende nullo l'effetto del gradiente
					end
				end
			end
		end
	end
end

-- CONTROLLI IN TEMPO REALE (Camera e Velocità)
rs.RenderStepped:Connect(function()
	if active then
		-- A. Anti-Rotate Camera (Raddrizza la visuale istantaneamente)
		local cam = workspace.CurrentCamera
		if cam then
			local look = cam.CFrame.LookVector
			local pos = cam.CFrame.Position
			cam.CFrame = CFrame.lookAt(pos, pos + look)
		end
		
		-- B. Anti-Slow (Controlla se la velocità scende sotto il normale)
		local char = lp.Character
		if char and char:FindFirstChild("Humanoid") then
			if char.Humanoid.WalkSpeed < 16 then
				char.Humanoid.WalkSpeed = 16
			end
		end
	end
end)

btn.MouseButton1Click:Connect(function()
	active = not active
	btn.Text = active and "NoLoveTrap: ON" or "NoLoveTrap: OFF"
	btn.BackgroundColor3 = active and Color3.fromRGB(0, 180, 0) or Color3.fromRGB(200, 0, 100)
	if active then cleanLove() else lt.FogEnd = 1000 end
end)

task.spawn(function()
	while true do
		if active then pcall(cleanLove) end
		task.wait(1) -- Pulizia GUI ogni secondo per non laggare
	end
end)
